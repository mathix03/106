DROP TABLE IF EXISTS t_livraison;

CREATE TABLE t_livraison (
    livraison_id INTEGER PRIMARY KEY,
    commande_fk INTEGER NOT NULL,
    livreur_fk VARCHAR(10) NOT NULL,
    statut ENUM('livree', 'annulee') NOT NULL,
    heure_depart DATETIME NOT NULL,
    heure_arrivee DATETIME NOT NULL
);
